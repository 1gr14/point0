import { Link } from '@/lib/navigate'
import { routes } from '@/lib/routes'
import { root } from '@/lib/root'
import { AppError } from '@/lib/error'
import { prisma } from '@/lib/prisma'
import { generalLayout } from '@/layouts/general'

export const page = generalLayout
  .lets('page', 'about', '/about')
  .loader(async () => {
    const lastIdea = await prisma.idea.findFirst({
      orderBy: {
        id: 'desc',
      },
    })
    if (!lastIdea) {
      throw new AppError('Last Idea Not Found :–(')
    }
    return {
      lastIdea: {
        id: lastIdea.id,
        title: lastIdea.title,
      },
    }
  })
  .page((props) => <MDXContent {...props} />)

<section className="space-y-6">
  <div className="space-y-2">
    <h1 className="text-3xl font-bold tracking-tight text-slate-900">About IdeaNick</h1>
    <p className="text-slate-600">IdeaNick is a platform for creating and sharing ideas.</p>
    <p className="text-slate-600">You can create ideas and add news updates to each one.</p>
  </div>

  <div className="rounded-xl border border-slate-200 bg-white p-4 shadow-sm">
    <p className="text-sm text-slate-500">Fresh idea</p>
    <p className="mt-1">
      <Link
        className="font-semibold text-cyan-700 hover:text-cyan-600"
        route="ideaView"
        input={{ id: props.data.lastIdea.id }}
      >
        {props.data.lastIdea.title}
      </Link>
    </p>
  </div>

  <Link
    className="inline-flex rounded-lg bg-slate-900 px-4 py-2 text-sm font-medium text-white hover:bg-slate-700"
    route="home"
  >
    Start Creating Ideas
  </Link>
</section>
